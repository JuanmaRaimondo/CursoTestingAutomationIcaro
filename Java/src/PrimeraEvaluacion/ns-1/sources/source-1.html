 package ClaseSeis.Excepciones;
 
 public class Calculadora {
 
     public double calcular(double numerouno, double numerodos, char operacion){
         double resultado = 0.0;
         switch (operacion){
             case '+':
                     resultado = numerouno + numerodos;
                 break;
             case '-':
                     resultado = numerouno - numerodos;
             break;
             case '*':
                     resultado = numerouno * numerodos;
             break;
             case '/':
                 if(numerodos != 0){
                     resultado = numerouno / numerodos;
                 }else {
                     throw new RuntimeException("La division por cero no esta definida, ingrese un valor valido");
                 }
             break;
             default:
                     throw new RuntimeException("No ingreso una operacion valida");
         }
         return resultado;
     }
 }
